3pub fn func_stock_var(ftp_result: &mut FtpResult, rownum: usize, colnum: usize, ncols: usize) {
4    let m = match &ftp_result.stock_amort {
5        Some(stock_amort) => stock_amort,
6        None => {
7            eprintln!("stock_amort is None, cannot read value.");
8            return;
9        }
10    };
11    // println!("ok");
12    // Check if stock_amort is Some and mutate it
13    if let Some(varstock_amort) = &mut ftp_result.varstock_amort {
14        if rownum == 0 || colnum == ncols - 1 {
15            varstock_amort[[rownum, colnum]] = m[[rownum, colnum]];
16        } else {
17            varstock_amort[[rownum, colnum]] = m[[rownum, colnum]] - m[[rownum - 1, colnum + 1]];
18        }
19    } else {
20        // Handle the case where stock_amort is None, if necessary
21        eprintln!("varstock_amort is None, cannot update value.");
22    }
23}
