3fn max_zero(a: &f64) -> f64 {
4    if *a < 0.0 {
5        0.0
6    } else {
7        *a
8    }
9}
10pub fn flux_func_stock_var(ftp_result: &mut FtpResult, rownum: usize, colnum: usize) {
11    let v_oust = &ftp_result.input_outstanding;
12
13    let m_profile = &ftp_result.input_profiles;
14
15    println!("ok");
16    // Check if stock_amort is Some and mutate it
17    if let Some(varstock_amort) = &mut ftp_result.varstock_amort {
18        if rownum == 0 {
19            varstock_amort[[rownum, colnum]] = m_profile[[rownum, colnum]] * v_oust[[rownum, 0]];
20        } else if colnum == 0 {
21            let mut front_amt: f64 = 0.0;
22            for i in 1..(rownum + 1) {
23                front_amt += varstock_amort[[rownum - i, i]];
24            }
25            front_amt = v_oust[[rownum, 0]] - front_amt;
26            varstock_amort[[rownum, colnum]] = max_zero(&front_amt);
27        } else {
28            varstock_amort[[rownum, colnum]] =
29                varstock_amort[[rownum, 0]] * m_profile[[rownum, colnum]];
30        }
31    } else {
32        // Handle the case where stock_amort is None, if necessary
33        eprintln!("varstock_amort is None, cannot update value.");
34    }
35}
