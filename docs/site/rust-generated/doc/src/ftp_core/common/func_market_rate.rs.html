3pub fn func_market_rate(ftp_result: &mut FtpResult, rownum: usize, colnum: usize, ncols: usize) {
4    let m_stock_instal = match &ftp_result.stock_instal {
5        Some(m_stock_instal) => m_stock_instal,
6        None => {
7            eprintln!("m_stock_instal is None, cannot read value.");
8            return;
9        }
10    };
11
12    let m_ftp_rate = match &ftp_result.ftp_rate {
13        Some(m_ftp_rate) => m_ftp_rate,
14        None => {
15            eprintln!("m_varstock_instal is None, cannot read value.");
16            return;
17        }
18    };
19
20    let m_input_rate = &ftp_result.input_rate;
21
22    if let Some(market_rate) = &mut ftp_result.market_rate {
23        if colnum == ncols - 1 {
24            market_rate[[rownum, colnum]] = m_input_rate[[rownum, colnum - 1]];
25        } else {
26            // colnum < ncols - 1
27            let a = m_ftp_rate[[rownum, colnum - 1]];
28            let mut b = 0.0;
29            let mut c = 0.0;
30            let d = m_stock_instal[[rownum, colnum]];
31
32            for k in colnum..ncols {
33                b += m_stock_instal[[rownum, k]];
34            }
35
36            for k in colnum + 1..ncols {
37                c += m_stock_instal[[rownum, k]] * market_rate[[rownum, k]];
38            }
39
40            if d != 0.0 {
41                market_rate[[rownum, colnum]] = ((a * b) - c) / d;
42            } else {
43                market_rate[[rownum, colnum]] = 0.0;
44            }
45        }
46    } else {
47        // Handle the case where varstock_amort is None, if necessary
48        eprintln!("market_rate is None, cannot update value.");
49    }
50}
